<script>

const messages=[

"✨ You make ordinary days feel special.",

"❤️ And aapke sath gediii toh bass... one of the thing I wait for every day.",

"🌸 Talking to you slowly became one of my favourite parts of every day.",

"🐱 Even the cutest cats would agree that you're adorable.",

"💖 I never wanted you to think you didn't matter to me.",

"🌷 Your smile always made my day a little brighter.",

"🥹 Thank you for being yourself. That's already more than enough.",

"❤️ I hope one day, this page makes you smile instead of overthink."

];

let current=0;

const card=document.getElementById("card");

const count=document.getElementById("count");

function update(){

card.style.opacity=0;

setTimeout(()=>{

card.innerHTML=messages[current];

count.innerHTML=(current+1)+" / "+messages.length;

card.style.opacity=1;

},180);

}

function nextCard(){

if(current<messages.length-1){

current++;

update();

}

}

function previousCard(){

if(current>0){

current--;

update();

}

}
