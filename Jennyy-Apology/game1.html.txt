<script>

const gameArea = document.getElementById("gameArea");
const scoreText = document.getElementById("score");
const message = document.getElementById("message");
const nextBtn = document.getElementById("nextBtn");

let score = 0;

const items = [

{emoji:"❤️",points:1},
{emoji:"💖",points:2},
{emoji:"🌹",points:3},
{emoji:"💌",points:0},
{emoji:"✨",points:2}

];

const loveMessages = [

"You make my days brighter ❤️",

"I hope you're smiling while playing this 🥹",

"You really mean a lot to me.",

"Thank you for giving this website a chance.",

"You're one of my favourite people ❤️",

"And aapke sath gediii toh bass... one of the thing I wait for every day ❤️"

];

function spawnItem(){

const item=document.createElement("div");

item.className="item";

const random=items[Math.floor(Math.random()*items.length)];

item.innerHTML=random.emoji;

item.dataset.points=random.points;

item.style.left=Math.random()*85+"%";

item.style.top="-40px";

gameArea.appendChild(item);

let y=-40;

const fall=setInterval(()=>{

y+=3;

item.style.top=y+"px";

if(y>gameArea.clientHeight){

clearInterval(fall);

item.remove();

}

},25);

item.onclick=function(){

clearInterval(fall);

item.remove();

if(random.emoji=="💌"){

message.innerHTML=loveMessages[Math.floor(Math.random()*loveMessages.length)];

}else{

score+=random.points;

scoreText.innerHTML=score;

message.innerHTML="+"+random.points+" points ❤️";

}

if(score>=20){

winGame();

}

};

}function winGame(){

message.innerHTML=

"🎉 Congratulations! You collected all my little feelings ❤️";

nextBtn.style.display="block";

createCelebration();

}

function createCelebration(){

for(let i=0;i<50;i++){

const heart=document.createElement("div");

heart.innerHTML="💖";

heart.style.position="absolute";

heart.style.left=Math.random()*100+"%";

heart.style.top=Math.random()*100+"%";

heart.style.fontSize=(20+Math.random()*20)+"px";

heart.style.pointerEvents="none";

heart.style.opacity="1";

heart.style.transition="all 2s ease";

gameArea.appendChild(heart);

setTimeout(()=>{

heart.style.transform=
"translateY(-80px) scale(1.8)";

heart.style.opacity="0";

},100);

setTimeout(()=>{

heart.remove();

},2000);

}

}

setInterval(spawnItem,700);

</script>
