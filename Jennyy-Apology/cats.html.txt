<script>

const messages = [

"❤️ He wanted you to know that you really matter to him.",

"🐱 This little cat says: 'I hope you smile today.'",

"🌸 Thank you for taking the time to read all of this.",

"💖 Even this cat knows you're someone really special.",

"🥹 If this cat could talk, it would ask for one more chance to make you smile.",

"✨ I hope one day this page becomes a happy memory instead of a sad one."

];

let clicked = new Set();

function showMessage(index, element){

document.getElementById("message").innerHTML = messages[index];

clicked.add(index);

element.style.transform = "scale(1.08)";
element.style.background = "#ffe6f2";

createHeart(element);

if(clicked.size==6){

document.getElementById("nextBtn").style.display="block";

document.getElementById("message").innerHTML =
"🎉 Yay! You found every kitty's message. Thank you for reading them all ❤️";

}

}

function createHeart(card){

const heart=document.createElement("div");

heart.innerHTML="💖";

heart.style.position="absolute";
heart.style.fontSize="24px";
heart.style.pointerEvents="none";

const rect=card.getBoundingClientRect();

heart.style.left=(rect.left+rect.width/2)+"px";
heart.style.top=(rect.top+rect.height/2)+"px";

heart.style.transition="all 1s ease";

document.body.appendChild(heart);

setTimeout(()=>{

heart.style.transform="translateY(-80px) scale(1.8)";
heart.style.opacity="0";

},50);

setTimeout(()=>{

heart.remove();

},1100);

}

</script>
