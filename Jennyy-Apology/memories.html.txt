<script>

const messages = [

"✨ You make ordinary days feel special.",

"❤️ And aapke sath gediii toh bass... one of the thing I wait for every day.",

"🌸 Talking to you became one of my favourite parts of every day.",

"💖 I never wanted you to feel like you didn't matter to me.",

"🥹 Thank you for being yourself. That's one of the reasons you're special.",

"🌷 I still hope I can make you smile again someday.",

"⭐ Some people become important without even realizing it.",

"💕 Your smile always made my day brighter.",

"🌙 No matter what happens, I'll always appreciate the moments we shared.",

"❤️ Thank you for reading this far."
];

let foundStars = new Set();

function showStar(index, star){

    document.getElementById("message").innerHTML = messages[index];

    if(!foundStars.has(index)){

        foundStars.add(index);

        star.style.opacity = "0.35";
        star.style.transform = "scale(1.4)";
        createSparkle(star);

    }

    if(foundStars.size === messages.length){

        document.getElementById("message").innerHTML =
        "🐱 You found every star! Thank you for reading every little message. ❤️";

        document.getElementById("nextBtn").style.display = "inline-block";
    }

}

function createSparkle(star){

    const rect = star.getBoundingClientRect();

    for(let i=0;i<8;i++){

        const sparkle = document.createElement("div");

        sparkle.innerHTML = "✨";

        sparkle.style.position = "fixed";
        sparkle.style.left = (rect.left + rect.width/2) + "px";
        sparkle.style.top = (rect.top + rect.height/2) + "px";
        sparkle.style.pointerEvents = "none";
        sparkle.style.transition = "all .8s ease";
        sparkle.style.fontSize = "20px";

        document.body.appendChild(sparkle);

        setTimeout(()=>{

            sparkle.style.transform =
            `translate(${(Math.random()*120)-60}px, ${(Math.random()*120)-60}px) scale(1.6)`;

            sparkle.style.opacity = "0";

        },30);

        setTimeout(()=>{

            sparkle.remove();

        },900);

    }

}
